Table User {
  id serial [pk, increment]
  name varchar(70)  [not null]
  email text [not null, unique]
  phone_number text [not null, unique]
  password text [not null]
  img_url text [not null, unique]
  role_id integer [not null, ref: > Role.id]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: `now()`, not null]
  updated_at timestamp [default: `now()`, not null]
  Indexes{
     email
     id
  }
}

Table Role {
  id serial [pk, increment]
  name varchar(70)  [not null, unique]
  Indexes{
     id
  }
}

Table Course {
  id serial [pk, increment]
  name varchar(70)  [not null, unique]
  user_id integer [not null, ref: > User.id]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: `now()`, not null]
  updated_at timestamp [default: `now()`, not null]
  Indexes{
     id
  }
}

Table Course_Member {
  id serial [pk, increment]
  user_id integer [not null, ref: > User.id]
  course_id integer [not null, ref: > Course.id]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: `now()`, not null]
  updated_at timestamp [default: `now()`, not null]
  Indexes{
     (user_id, course_id) [unique]
  }
}


Table Exam {
  id serial [pk, increment]
  name varchar(70)  [not null, unique]
  course_id integer [not null, ref: > Course.id]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: `now()`, not null]
  updated_at timestamp [default: `now()`, not null]
  Indexes{
     id
  }
}

Table Qs_Type {
  id serial [pk, increment]
  name varchar(70)  [not null, unique]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: `now()`, not null]
  updated_at timestamp [default: `now()`, not null]
  Indexes{
     id
  }
}

Table Qs {
  id serial [pk, increment]
  question text [not null, unique]
  exam_id integer [not null, ref: > Exam.id]
  question_type_id integer [not null, ref: > Qs_Type.id, not null]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: `now()`, not null]
  updated_at timestamp [default: `now()`, not null]
  Indexes{
     (id, exam_id) [unique]
  }
}

Table Choices {
  id serial [pk, increment]
  question_id text [not null, unique, ref: > Qs.id]
  choice_a text [not null]
  choice_b text [not null]
  choice_c text [not null]
  choice_d text [not null]
  correct_ans char [not null] 
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: `now()`, not null]
  updated_at timestamp [default: `now()`, not null]
  Indexes{
     (id, question_id) [unique]
  }
}

Table TF_Answers {
  id serial [pk, increment]
  question_id text [not null, unique, ref: > Qs.id, unique]
  ans boolean [not null]
  created_at timestamp [default: `now()`, not null]
  deleted_at timestamp [default: `now()`, not null]
  updated_at timestamp [default: `now()`, not null]
  Indexes{
     (id, question_id) [unique]
  }
}
